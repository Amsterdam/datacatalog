[docs]def _serviceType_mapping(data: dict) -> T.Optional[str]:
    if data['resource_type'] != 'api':
        return None
    ckan_format = data.get('format', '').lower()
    return ckan_format if ckan_format in ('wms', 'wfs') else 'other'
